Use these settings as the deployment contract for a self-hosted 143 environment. Exact names may vary by template, but the categories should remain stable across API, frontend, worker, database, Redis, GitHub, and LLM configuration.

## Application

<ConfigField name="IMAGE_REGISTRY" type="string" description="Container registry prefix for runtime images. Defaults to ghcr.io/assembledhq in the single-node compose file." />
<ConfigField name="IMAGE_TAG" type="string" description="Runtime image tag or release SHA used by the single-node compose file." />
<ConfigField name="DOMAIN" type="string" description="Public domain served by Caddy in the single-node deployment." />
<ConfigField name="BASE_URL" type="url" description="Public backend/app origin used for callbacks and internal links." />
<ConfigField name="FRONTEND_URL" type="url" description="Public frontend origin used for browser redirects." />
<ConfigField name="CORS_ALLOWED_ORIGINS" type="string" description="Comma-separated browser origins allowed to call the API." />
<ConfigField name="PREVIEW_ORIGIN_TEMPLATE" type="string" description="Preview wildcard URL template, such as https://{id}.preview.example.com." />
<ConfigField name="CLOUDFLARE_API_TOKEN" type="secret" description="Cloudflare DNS token used by the bundled single-node Caddy wildcard certificate flow." />
<ConfigField name="SESSION_SECRET" type="secret" description="Secret used for session signing." />
<ConfigField name="CSRF_SIGNING_KEY" type="secret" description="Secret used for CSRF token signing in production." />
<ConfigField name="ENCRYPTION_MASTER_KEY" type="secret" description="Master key for encrypted credentials at rest." />
<ConfigField name="DATABASE_URL" type="secret" description="Postgres connection string." />
<ConfigField name="REDIS_URL" type="secret" description="Redis connection string." />

## GitHub

<ConfigField name="GITHUB_APP_ID" type="string" description="GitHub App ID." />
<ConfigField name="GITHUB_OAUTH_CLIENT_ID" type="string" description="GitHub OAuth client ID." />
<ConfigField name="GITHUB_OAUTH_CLIENT_SECRET" type="secret" description="GitHub OAuth client secret." />
<ConfigField name="GITHUB_WEBHOOK_SECRET" type="secret" description="Secret used to verify GitHub webhooks." />
<ConfigField name="GITHUB_APP_PRIVATE_KEY" type="secret" description="Private key for the GitHub App." />

## Workers

<ConfigField name="NODE_ID" type="string" description="Stable node identifier used for worker heartbeats and preview routing." />
<ConfigField name="WORKER_PROCESS_COUNT" type="integer" description="Worker loops per host." />
<ConfigField name="WORKER_MAX_ACTIVE_SANDBOXES" type="integer" description="Maximum live sandboxes on a worker host." />
<ConfigField name="SESSION_EXECUTOR_IMAGE" type="string" description="Server image used for durable session-executor containers in production worker-capable modes." />
<ConfigField name="SESSION_EXECUTOR_DOCKER_NETWORK" type="string" description="Docker network session executors join to reach Postgres and worker APIs." />
<ConfigField name="SESSION_EXECUTOR_EXTRA_BINDS" type="string" description="Comma-separated extra host bind mounts for session executors, such as /var/lib/143:/var/lib/143 on single-node installs." />
<ConfigField name="DOCKER_GID" type="string" description="Host docker group id passed to worker and executor containers for docker.sock access." />
<ConfigField name="SANDBOX_IMAGE" type="string" description="Sandbox runtime image used by worker-created sandbox containers." />
<ConfigField name="SANDBOX_RUNTIME" type="string" description="Docker runtime for sandboxes, usually runsc in production." />
<ConfigField name="SANDBOX_REQUIRE_GVISOR" type="boolean" description="Fail worker startup when gVisor is unavailable." />
<ConfigField name="SANDBOX_REQUIRE_DISK_QUOTA" type="boolean" description="Fail or allow startup when Docker cannot enforce sandbox rootfs disk quotas." />
<ConfigField name="SANDBOX_CPU_LIMIT" type="string" description="Sandbox CPU limit." />
<ConfigField name="SANDBOX_MEMORY_LIMIT_MB" type="integer" description="Sandbox memory limit." />
<ConfigField name="SANDBOX_DISK_LIMIT_GB" type="integer" description="Sandbox disk limit." />

## Single-node storage

<ConfigField name="SINGLE_NODE_DATA_DIR" type="path" description="Host-backed durable data root mounted into the API/worker container. Defaults to /var/lib/143." />
<ConfigField name="HOST_143_DIR" type="path" description="Host directory mounted read-only at /etc/143 for sandbox resolver and capability files. Defaults to /etc/143." />
<ConfigField name="SANDBOX_RESOLV_CONF" type="path" description="Resolver config file passed to sandbox containers. Defaults to /etc/143/sandbox-resolv.conf." />
<ConfigField name="SANDBOX_AUTH_SOCKET_DIR" type="path" description="Host directory for the worker-owned sandbox credential socket. Defaults to /var/run/143/sandbox-auth and must be 0750 or stricter." />
<ConfigField name="PREVIEW_DEPENDENCY_CACHE_LOCAL_DIR" type="path" description="Host-backed local preview dependency cache. Defaults to /var/cache/143/preview-dependency-cache." />

<Callout type="warning" title="Secret handling">
Production secrets should live in the deployment's encrypted secret system. Do not commit plaintext `.env` files.
</Callout>
